drop table if exists falkutet;

create table falkutet(
	id int auto_increment not null,
    naziv varchar(40) not null,
    god_osnivanja int not null,
    primary key(id)
);

drop table if exists student;

create table student(
	id int auto_increment not null,
    ime varchar(40) not null,
    prezime varchar(40) not null,
    god_upisa int not null,
    studentId int not null,
    foreign key(studentId) references falkutet(id),
    primary key(id)
);
drop table if exists kurs;

create table kurs(
	id int auto_increment not null,
    naziv varchar(40) not null,
    broj_ESPB int not null,
    kursId int not null,
    foreign key(kursId) references falkutet(id),
    primary key(id)
);

drop table if exists student_kurs;

create table student_kurs(
	id int auto_increment not null,
	studentId int not null,
	kursId int not null,
    foreign key(studentId) references student(id),
	foreign key(kursId) references kurs(id),
	primary key(id)

);
